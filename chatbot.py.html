def chatbot():
    print("Hola! Soy tu asistente virtual")
    print("Puedes preguntarme sobre: funcionamiento del KDAI, precio, detalles, o escribir 'salir' para terminar.")

    while True:
        pregunta = input("Pregunta: ").lower()

        if "hace" in pregunta or "funciona" in pregunta:
            print("Bot: El KDAI es un kit de aprendizaje con diversas actividades para desarrollar las habilidades cognitivas de tus niños")

        elif "precio" in pregunta or "cuesta" in pregunta or "vale" in pregunta:
            print("Bot: El precio del KDAI es de $100.")

        elif "incluye" in pregunta :
            print("Bot: El KDAI incluye juegos de memoria, de reconocimiento, de lógica, rítmicos y de rompecabezas")
            print("Las actividades son las siguientes: Juego del xilófono, Pares iguales, Rompecabezas , Pizarra interactiva, Verdadero y falso")


        elif "características" in pregunta or "detalles" in pregunta:
            print("Bot: El KDAI es una caja de 20x20 centímetros, e incluye 5 actividades recreacionales y educativas para los niños.")
            print("Se recomienda para niños de 5 a 7 años.")

        elif "salir" in pregunta:
            print("Bot: ¡Gracias por tu interés! Hasta pronto.")
            break

        else:
            print("Bot: Lo siento, no entendí tu pregunta. Puedes preguntarme sobre funcionamiento, precio o detalles.")

if __name__ == "__main__":
    chatbot()
